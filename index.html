#include<iostream>
using namespace std;

class Circle
{
private:
    double r;
public:
    void Setr(double r1);
    double Area();
    double Perimeter();
    Circle()
    {
    r = 0;
    }
    Circle(double r0)
    {
    r = r0;
    }
    Circle(Circle &robj)
    {
    r = robj.r;
    }
};
void Circle::Setr(double r1)
{
    r = r1;
}
double Circle::Area()
{
    return r * r * 3.14;
}
double Circle::Perimeter()
{
    return r * 2 * 3.14;
}

int main()
{
    Circle c1;
    int i;
    cout << "请输入半径长度" << endl;
    cin >> i;
    c1.Setr(i);
    cout << c1.Area() << endl;
    cout << c1.Perimeter() << endl;
    Circle c2(2 * i);
    cout << c2.Area() << endl;
    cout << c2.Perimeter() << endl;
    Circle c3(c1);
    cout << c3.Area() << endl;
    cout << c3.Perimeter() << endl;
    return 0;
}
